
-- Create an entity table to store customer, vendor, and employee information
CREATE TABLE entity (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  type STRING NOT NULL,
  name STRING NOT NULL,
  email STRING NOT NULL,
  phone STRING,
  address STRING,
  city STRING,
  state STRING,
  zip STRING
);
